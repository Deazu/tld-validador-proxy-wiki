Identidad y Canal (DynamoDB)

Auditoría: canal.js (legado) y proxy autónomo

En tld-api-cuenta-nombre/.../canal.js, getCanal(id) hace Query a tld-validador-canal por idCanal; si no hay resultado, Scan filtrado por alias. Sobre el ítem: resuelve llaveCifrado y llaveDescifrado vía getLlave cuando el valor no es ARN; carga getOperaciones desde tld-validador-canal-operacion. Campos que el motor de negocio espera hoy incluyen algoritmoCifrado, llaveCifrado, operaciones y metadatos de canal. El proxy autónomo debe añadir urlValidador (no presente en el legado analizado) para el remote endpoint. 

flowchart TD
    A[id o alias] --> B{Query idCanal}
    B -->|hit| C[unmarshall item]
    B -->|miss| D[Scan por alias]
    D --> C
    C --> E{llave es ARN?}
    E -->|si| F[origen KMS string]
    E -->|no| G[getLlave idLlave]
    G --> H[registro archivo EFS]
    C --> I[getOperaciones]
    I --> J["DTO inmutable del proxy"]

operacionesPaquete.js se usa tal cual en el proxy para cifrar: cerrarPaquete(origen, contenido, modo) con modo = algoritmoCifrado leído de DynamoDB, nunca del payload del invocador. Ver Cifrado (aplicación). Logs: Logger con PRINT_LOGS y sanitize para no volcar llaves ni PII en CloudWatch.

La tabla tld-validador-canal no es “solo validadores”

Almacena filas de canal usadas en dos papeles en el mismo flujo: emisor (cliente del API) y validador (destino remoto del target). El rol lo define el contexto, no el nombre de la tabla. El proxy usa principalmente idCanalValidador para resolver destino HTTPS, llaves y algoritmo.

  • tld-validador-canal-llave — referencias idLlave, archivo bajo EFS o ARN KMS.
  • tld-validador-canal-operacion — operaciones habilitadas por metodo.
  • tld-validador-canal-auth-token — tokens para auth dinámica hacia el remote endpoint (si aplica).

Modelo de datos ideal — tld-validador-canal

Contrato orientado al proxy autónomo. Los tipos son orientativos; alinear con el esquema real en DynamoDB.

Atributo Obligatorio (proxy) Descripción
idCanal Sí Clave de partición lógica; identificador estable del canal.
alias No Resolución alternativa si el cliente envía alias en lugar de id (opcional según política).
urlValidador Sí (nuevo) URL HTTPS del remote endpoint (p. ej. https://validador.ejemplo/api/ruta). Sin esto el proxy no puede enrutar el tráfico de forma autónoma.
llaveCifrado Sí ARN KMS (string con prefijo arn) o idLlave a resolver en tld-validador-canal-llave para cerrarPaquete.
llaveDescifrado Según flujo Necesaria si el proxy abre respuestas cifradas hacia el emisor; para solo “cifrar y enviar” puede no usarse en el camino feliz.
algoritmoCifrado Sí Modo simétrico Node (aes-256-gcm, aes-256-cbc, …) pasado a operacionesPaquete.cerrarPaquete.
estado Recomendado Si existe, el proxy debe rechazar canales inactivos antes de cifrar.

Propuesta de código — getCanal para el Proxy

Patrón: sin mutación in-place del ítem crudo de Dynamo; errores explícitos; logs solo vía Print con metadatos no sensibles (sin volcar Items completos). 

'use strict';

const { DynamoDBClient, GetItemCommand, QueryCommand, ScanCommand } = require('@aws-sdk/client-dynamodb');
const { marshall, unmarshall } = require('@aws-sdk/util-dynamodb');
const { Print } = require('./logger'); // logger.js refactory

const client = new DynamoDBClient({});
const TABLA_CANAL = 'tld-validador-canal';
const TABLA_CANAL_LLAVE = 'tld-validador-canal-llave';
const TABLA_CANAL_OPERACION = 'tld-validador-canal-operacion';

class CanalNotFound extends Error {
  constructor(id) {
    super(`Canal no encontrado: ${id}`);
    this.name = 'CanalNotFound';
    this.code = 'CANAL_NOT_FOUND';
  }
}

class CanalDataError extends Error {
  constructor(message, cause) {
    super(message);
    this.name = 'CanalDataError';
    this.code = 'CANAL_DATA_ERROR';
    if (cause) this.cause = cause;
  }
}

/** Resuelve idLlave → registro; no loguear el ítem completo. */
async function fetchLlaveRegistro(idLlave) {
  const resp = await client.send(
    new GetItemCommand({
      TableName: TABLA_CANAL_LLAVE,
      Key: marshall({ idLlave }),
    })
  );
  if (!resp.Item) throw new CanalDataError(`Llave inexistente: ${idLlave}`);
  return unmarshall(resp.Item);
}

/**
 * Convierte referencia de canal (ARN o id de llave) al "origen" de cerrarPaquete.
 * - ARN: devuelve el string.
 * - id: devuelve { archivo } según registro en TABLA_CANAL_LLAVE.
 */
async function resolveOrigenLlave(ref) {
  if (typeof ref !== 'string' || !ref) {
    throw new CanalDataError('Referencia de llave vacía');
  }
  if (ref.startsWith('arn')) return ref;
  const reg = await fetchLlaveRegistro(ref);
  if (!reg.archivo) throw new CanalDataError(`Llave ${ref} sin campo archivo`);
  return { archivo: reg.archivo };
}

async function fetchOperaciones(idCanal) {
  const resp = await client.send(
    new QueryCommand({
      TableName: TABLA_CANAL_OPERACION,
      KeyConditionExpression: 'idCanal = :id',
      ExpressionAttributeValues: marshall({ ':id': idCanal }),
    })
  );
  return (resp.Items || []).map((item) => unmarshall(item));
}

/** Fila cruda → DTO inmutable para el handler. */
async function buildCanalDto(raw) {
  if (!raw.urlValidador) {
    throw new CanalDataError(`Canal ${raw.idCanal}: falta urlValidador`);
  }
  const origenCifrado = await resolveOrigenLlave(raw.llaveCifrado);
  let origenDescifrado = null;
  if (raw.llaveDescifrado) {
    origenDescifrado = await resolveOrigenLlave(raw.llaveDescifrado);
  }
  const operaciones = await fetchOperaciones(raw.idCanal);
  return Object.freeze({
    idCanal: raw.idCanal,
    alias: raw.alias,
    urlValidador: raw.urlValidador,
    algoritmoCifrado: raw.algoritmoCifrado,
    origenCifrado,
    origenDescifrado,
    operaciones,
    rawEstado: raw.estado,
  });
}

async function queryCanalPorId(idCanal) {
  const resp = await client.send(
    new QueryCommand({
      TableName: TABLA_CANAL,
      KeyConditionExpression: 'idCanal = :id',
      ExpressionAttributeValues: marshall({ ':id': idCanal }),
    })
  );
  if (!resp.Items?.length) return null;
  return unmarshall(resp.Items[0]);
}

async function scanCanalPorAlias(alias) {
  const resp = await client.send(
    new ScanCommand({
      TableName: TABLA_CANAL,
      FilterExpression: 'alias = :a',
      ExpressionAttributeValues: marshall({ ':a': alias }),
    })
  );
  if (!resp.Items?.length) return null;
  return unmarshall(resp.Items[0]);
}

/**
 * API principal del proxy: siempre devuelve DTO o lanza.
 */
async function getCanal(idOCAlias) {
  Print('[canal] resolviendo', { idOCAlias });
  let row = await queryCanalPorId(idOCAlias);
  if (!row) row = await scanCanalPorAlias(idOCAlias);
  if (!row) throw new CanalNotFound(idOCAlias);
  return buildCanalDto(row);
}

/** ¿El método está habilitado para este canal? */
function metodoDisponible(dto, metodo) {
  if (!dto.operaciones || dto.operaciones.length === 0) return false;
  return dto.operaciones.some(
    (o) => o.operacion === metodo && String(o.estado).toUpperCase() === 'Y'
  );
}

module.exports = {
  getCanal,
  metodoDisponible,
  CanalNotFound,
  CanalDataError,
};

Corrección de metodoDisponible

El código legado en tld-api-cuenta-nombre contenía el typo lenght y una condición incorrecta sobre el array vacío. La versión propuesta usa .length, comprueba ausencia de operaciones y valida estado de forma explícita (Y en mayúsculas tras normalizar).

Evolución de clase: IdentityManager (contrato proxy)

Proyección formal sobre el legado canal.js: elimina Scan en caliente sustituyéndolo por GSI alias-index, usa exclusivamente AWS SDK v3 (@aws-sdk/client-dynamodb), y trata “canal no encontrado” como error explícito (CanalNotFoundError) en lugar de null silencioso. El DTO devuelto es inmutable (Object.freeze en profundidad) para que ningún paso posterior del handler altere accidentalmente la configuración del canal (análogo a no permitir que alguien reescriba el pasaporte en mitad del control del portero).

Matriz de errores: PROXY_ERR_ID_001. Orquestación: app.js.

lambdas/proxy/lib/IdentityManager.js (propuesta) 
'use strict';

const {
  DynamoDBClient,
  QueryCommand,
  GetItemCommand,
} = require('@aws-sdk/client-dynamodb');
const { marshall, unmarshall } = require('@aws-sdk/util-dynamodb');

/**
 * Canal u otro lookup de identidad inexistente.
 * El handler traduce a HTTP (404 / 500 según matriz) y escribe telemetría con `detail` = `{ ...err.telemetryDetail, phase }`.
 */
class CanalNotFoundError extends Error {
  constructor(lookupKey, opts = {}) {
    super(opts.message ?? `Canal no encontrado: ${lookupKey}`);
    this.name = 'CanalNotFoundError';
    this.lookupKey = String(lookupKey);
    this.code = opts.internalCode ?? 'PROXY_ERR_ID_001';
    this.reason = opts.reason ?? 'CANAL_NOT_FOUND';
  }

  get telemetryDetail() {
    return {
      internalCode: this.code,
      reason: this.reason,
      lookupKey: this.lookupKey,
    };
  }
}

function deepFreeze(o) {
  if (o === null || typeof o !== 'object') return o;
  Object.freeze(o);
  for (const k of Object.keys(o)) {
    const v = o[k];
    if (v && typeof v === 'object') deepFreeze(v);
  }
  return o;
}

/**
 * Instancia inmutable: tras el constructor, no se añaden propiedades mutables.
 * Configuración congelada en _cfg (no enumerable).
 */
class IdentityManager {
  constructor(cfg) {
    const frozen = Object.freeze({
      tableCanal: cfg.tableCanal,
      tableLlave: cfg.tableLlave,
      tableOperacion: cfg.tableOperacion,
      aliasGsiName: cfg.aliasGsiName,
      client: cfg.client instanceof DynamoDBClient ? cfg.client : new DynamoDBClient({}),
    });
    Object.defineProperty(this, '_cfg', { value: frozen, enumerable: false });
    Object.freeze(this);
  }

  async resolveChannel(idOrAlias) {
    const { client, tableCanal, tableLlave, tableOperacion, aliasGsiName } = this._cfg;
    let row = await this.#queryCanalPorId(client, tableCanal, idOrAlias);
    if (!row) row = await this.#queryCanalPorAlias(client, tableCanal, aliasGsiName, idOrAlias);
    if (!row) throw new CanalNotFoundError(idOrAlias);

    const canal = unmarshall(row);
    await this.#hydrateLlaves(client, tableLlave, canal);
    canal.operaciones = await this.#loadOperaciones(client, tableOperacion, canal.idCanal);
    return deepFreeze(canal);
  }

  metodoDisponible(canalDto, metodo) {
    if (!canalDto.operaciones || canalDto.operaciones.length === 0) return false;
    return canalDto.operaciones.some(
      (o) => o.operacion === metodo && String(o.estado).toUpperCase() === 'Y'
    );
  }

  async #queryCanalPorId(client, table, id) {
    const out = await client.send(
      new GetItemCommand({
        TableName: table,
        Key: marshall({ idCanal: id }),
      })
    );
    return out.Item ?? null;
  }

  async #queryCanalPorAlias(client, table, indexName, alias) {
    if (!indexName) return null;
    const out = await client.send(
      new QueryCommand({
        TableName: table,
        IndexName: indexName,
        KeyConditionExpression: 'alias = :a',
        ExpressionAttributeValues: marshall({ ':a': String(alias).toLowerCase() }),
        Limit: 1,
      })
    );
    return out.Items?.[0] ?? null;
  }

  async #hydrateLlaves(client, tableLlave, canal) {
    for (const k of ['llaveCifrado', 'llaveDescifrado']) {
      const v = canal[k];
      if (typeof v === 'string' && v.length > 0 && !v.startsWith('arn:')) {
        canal[k] = await this.#getLlave(client, tableLlave, v);
      }
    }
  }

  async #getLlave(client, tableLlave, idLlave) {
    const out = await client.send(
      new GetItemCommand({
        TableName: tableLlave,
        Key: marshall({ idLlave }),
      })
    );
    if (!out.Item) {
      throw new CanalNotFoundError(idLlave, {
        internalCode: 'PROXY_ERR_ENC_001',
        reason: 'LLAVE_NOT_FOUND',
        message: `Llave no encontrada: ${idLlave}`,
      });
    }
    return unmarshall(out.Item);
  }

  async #loadOperaciones(client, tableOperacion, idCanal) {
    const out = await client.send(
      new QueryCommand({
        TableName: tableOperacion,
        KeyConditionExpression: 'idCanal = :id',
        ExpressionAttributeValues: marshall({ ':id': idCanal }),
      })
    );
    return (out.Items ?? []).map((item) => unmarshall(item));
  }
}

module.exports = { IdentityManager, CanalNotFoundError };
  • GetItem por id: con PK idCanal sin SK, la resolución por id es un único GetItemCommand (menos RTT que Query).
  • Llave ausente: mismo tipo de error con internalCode: PROXY_ERR_ENC_001 y reason: LLAVE_NOT_FOUND; el handler no debe ramificar strings, solo propagar err.telemetryDetail al emit de telemetría.
  • Inmutabilidad: deepFreeze impide mutar canal.operaciones.push(...) por accidente (nadie reescribe el pasaporte tras el portero); si alguna utilidad interna necesita copia mutable, clonar antes de congelar.
  • Telemetría: patrón recomendado en el orquestador: telemetry.emit({ correlationId, phase: 'Identity' | 'Crypto_App', outcome: 'FAIL', detail: { ...e.telemetryDetail } }) cuando e instanceof CanalNotFoundError.
  • Runtime: métodos privados # requieren Node.js 14+; optional chaining requiere entorno acorde al despliegue Lambda elegido.
